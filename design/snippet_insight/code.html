export function debounce<T extends (...args: any[]) => any>(
    func: T,
    wait: number
): (...args: Parameters<T>) => void {
    let timeoutId: ReturnType<typeof setTimeout> | null = null;

    return function this(this: any, ...args: Parameters<T>) {
        if (timeoutId !== null) {
            clearTimeout(timeoutId);
        }
        
        timeoutId = setTimeout(() => {
            func.apply(this, args);
        }, wait);
    };
}

AI Explanation

This implementation creates a robust debounce function in TypeScript. It ensures that the provided func is only executed after wait milliseconds have passed since the last time it was invoked.

  • Generics: Uses T to capture the precise signature of the input function, ensuring the returned function maintains full type safety for arguments.
  • Context Preservation: Utilizes apply(this, args) to ensure that the original execution context (this) is correctly preserved when the function finally fires.
